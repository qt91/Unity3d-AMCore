using UnityEngine;
using System.Collections;

public class ItemVietjets : MonoBehaviour {

    public Material Gau;
    public Material MocKhoa;
    public Material MocKhoa1;
    public Material Non;
    void Start () {
        foreach (var item in gameObject.GetComponentsInChildren<ItemVietjet>())
        {
            int rand = Random.Range(0, 100);
            if (rand < AMGlobal.Settings.percentItem)
            {
                rand = Random.Range(0, 100);
                int step1 = 0;
                int step2 = AMGlobal.Settings.percentMocGau;
                int step3 = step2 + AMGlobal.Settings.percentMocKhoa;
                int step4 = step3 + AMGlobal.Settings.percentMocKhoa1;
                int step5 = step4 + AMGlobal.Settings.percentNon;

                if (rand > step1 && rand < step2)
                {
                    item.gameObject.GetComponent<ItemVietjet>().Type = "Gau";
                    item.gameObject.GetComponent<ItemVietjet>().Origin.GetComponent<MeshRenderer>().material = Gau;
                }
                else
                {
                    if (rand > step2 && rand < step3)
                    {
                        item.gameObject.GetComponent<ItemVietjet>().Type = "MocKhoa";
                        item.gameObject.GetComponent<ItemVietjet>().Origin.GetComponent<MeshRenderer>().material = MocKhoa;
                    }
                    else
                    {
                        if (rand > step3 && rand < step4)
                        {
                            item.gameObject.GetComponent<ItemVietjet>().Type = "MocKhoa";
                            item.gameObject.GetComponent<ItemVietjet>().Origin.GetComponent<MeshRenderer>().material = MocKhoa1;
                        }
                        else
                        {
                            item.gameObject.GetComponent<ItemVietjet>().Type = "Non";
                            item.gameObject.GetComponent<ItemVietjet>().Origin.GetComponent<MeshRenderer>().material = Non;
                        }
                    }
                }
            }else
            {
                Destroy(item.gameObject);
            }
        }
	}
	
	// Update is called once per frame
	void Update () {
	
	}
}
